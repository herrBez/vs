\exercise{2.23}
{
  Mostrare che i seguenti \textit{statement} del linguaggio \textbf{While} sono
  semanticamente equivalenti:
  \begin{enumerate}
    \item ``$S$; skip'' and ``S''
    \item ``while $b$ do $S$'' and
      ``if $b$ then ($S$; while $b$ do $S$) else skip''
    \item ``$S_1$; ($S_2$; $S_3$)'' and ``($S_1$; $S_2$); $S_3$''
  \end{enumerate}
}
